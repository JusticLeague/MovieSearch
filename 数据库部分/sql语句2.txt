查询电影导演演员
CREATE VIEW moviedirectoractor AS 
SELECT 
m.movie_name,m.movie_type,m.movie_country,m.movie_date,m.movie_score,
p.poster_pic,
d.director_name,
a.actor_name
FROM
movie AS m,director AS d,movie_item AS mi,actor AS a,poster AS p
WHERE
m.director_id=d.director_id
AND
m.movie_id=mi.movie_id
AND
mi.actor_id=a.actor_id
AND
p.movie_id=m.movie_id;

select * from moviedirectoractor;


查询电影信息、导演、演员
CREATE VIEW movieactor AS 
SELECT 
m.movie_name,
a.actor_name
FROM
movie AS m,movieitem AS mi,actor AS a
WHERE
m.movie_id=mi.movie_id
AND
mi.actor_id=a.actor_id;

select * from movieactor;

查询某个电影的的所有演员
SELECT 
a.actor_name
FROM
actor as a, movieitem as mi, movie as m
WHERE 
m.movie_id = 1
AND
mi.movie_id=m.movie_id
AND
mi.actor_id = a.actor_id;


查询演员的电影

a.actor_name ,m.movie_name

movie as m, movieitem as mi, actor as a

a.actor_id = mi.actor_id
AND
mi.movie_id = m.movie_id;


create view director_info as
select
d.director_name,
d.director_country,
d.director_birth,
dp.director_pic,
m.movie_name,
d.director_desc
from
movie as m, director as d, director_pic as dp

m.director_id = d.director_id;
and
d.director_id = dp.director_id;
